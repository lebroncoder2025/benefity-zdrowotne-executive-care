%--- 1 ---
\noindent{\color{brand}\bfseries 1.~Ogolnopolski dostęp}
\begin{itemize}
  \item w pakiecie są zarowno małe gabinety z wysoko ocenianymi specjalistami, jak i duże renomowane placowki,
  \item ponad 3\,400 placowek w całej Polsce,
  \item możliwość włączenia placowki wskazanej przez pracownikow.
\end{itemize}

%--- 2 ---
\noindent{\color{brand}\bfseries 2.~Wygoda korzystania}
\begin{itemize}
  \item nielimitowany dostęp do specjalistow bez skierowania,
  \item lekarz POZ w ciągu 1 dnia roboczego, specjalista w ciągu 5 dni,
  \item e-wizyty: konsultacje, skierowania, e-recepty, e-zwolnienia i omawianie wynikow.
\end{itemize}

%--- 3 ---
\noindent{\color{brand}\bfseries 3.~Rehabilitacja i wsparcie po wypadkach}
\begin{itemize}
  \item rehabilitacja powypadkowa ze sprzętem w cenie pakietu,
  \item organizacja pomocy po hospitalizacji: pielęgniarka, pomoc domowa, zwrot kosztow lekow,
  \item dodatkowa rehabilitacja bolowa, często przydatna przy pracy biurowej i przeciążeniach kręgosłupa.
\end{itemize}

%--- 4 ---
\noindent{\color{brand}\bfseries 4.~Elastyczność i dopasowanie}
\begin{itemize}
  \item możliwość zmiany pakietu na wyższy w każdej chwili,
  \item częściowa refundacja kosztow z placowek spoza listy,
  \item model łatwo dopasować do potrzeb firmy i struktury zespołu.
\end{itemize}

%--- 5 ---
\noindent{\color{brand}\bfseries 5.~Pełna obsługa i opieka doradcy}
\begin{itemize}
  \item bezpośredni kontakt z doradcą na każdym etapie,
  \item wsparcie przy wdrażaniu, deklaracjach i bieżących pytaniach,
  \item organizacja akcji wellbeingowych i edukacyjnych, aby pracownicy faktycznie korzystali z pakietu.
\end{itemize}

%--- 6 ---
\noindent{\color{brand}\bfseries 6.~Pakiet Rodzic}
\begin{itemize}
  \item specjalny, dedykowany pakiet ze specjalistami dla starszych osob,
  \item rozwiązanie przydatne tam, gdzie pracownicy często wspierają rodzicow lub opiekunow.
\end{itemize}
