{\Large\bfseries \color{branddark} 6 kluczowych atutów}\par
\vspace{0.7em}

\textbf{1. Ogólnopolski dostęp}
\begin{itemize}
  \item w pakiecie są zarówno małe gabinety z wysoko ocenianymi specjalistami, jak i duże renomowane placówki,
  \item ponad 3400 placówek w całej Polsce,
  \item możliwość włączenia placówki wskazanej przez pracowników.
\end{itemize}

\textbf{2. Wygoda korzystania}
\begin{itemize}
  \item nielimitowany dostęp do specjalistów bez skierowania,
  \item lekarz POZ w ciągu 1 dnia roboczego, specjalista w ciągu 5 dni,
  \item e-wizyty: konsultacje, skierowania, e-recepty, e-zwolnienia i omawianie wyników.
\end{itemize}

\textbf{3. Rehabilitacja i wsparcie po wypadkach}
\begin{itemize}
  \item rehabilitacja powypadkowa ze sprzętem w cenie pakietu,
  \item organizacja pomocy po hospitalizacji: pielęgniarka, pomoc domowa, zwrot kosztów leków,
  \item dodatkowa rehabilitacja bólowa, często przydatna przy pracy biurowej i przeciążeniach kręgosłupa.
\end{itemize}

\textbf{4. Elastyczność i dopasowanie}
\begin{itemize}
  \item możliwość zmiany pakietu na wyższy w każdej chwili,
  \item częściowa refundacja kosztów z placówek spoza listy,
  \item model łatwo dopasować do potrzeb firmy i struktury zespołu.
\end{itemize}

\textbf{5. Pełna obsługa i opieka doradcy}
\begin{itemize}
  \item bezpośredni kontakt z doradcą na każdym etapie,
  \item wsparcie przy wdrażaniu, deklaracjach i bieżących pytaniach,
  \item organizacja akcji wellbeingowych i edukacyjnych, aby pracownicy faktycznie korzystali z pakietu.
\end{itemize}

\textbf{6. Pakiet Rodzic}
\begin{itemize}
  \item specjalny, dedykowany pakiet ze specjalistami dla starszych osób,
  \item rozwiązanie przydatne tam, gdzie pracownicy często wspierają rodziców lub opiekunów.
\end{itemize}
