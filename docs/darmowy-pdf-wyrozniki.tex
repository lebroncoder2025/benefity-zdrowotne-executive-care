\vspace{0.35em}

%--- 1 ---
\noindent{\color{brand}\bfseries 1.~Og\'olnopolski dost\k{e}p}
\begin{itemize}
  \item w pakiecie s\k{a} zar\'owno ma\l{}e gabinety z wysoko ocenianymi specjalistami, jak i du\.ze renomowane plac\'owki,
  \item ponad 3\,400 plac\'owek w ca\l{}ej Polsce,
  \item mo\.zliwo\'s\'c w\l{}\k{a}czenia plac\'owki wskazanej przez pracownik\'ow.
\end{itemize}

%--- 2 ---
\noindent{\color{brand}\bfseries 2.~Wygoda korzystania}
\begin{itemize}
  \item nielimitowany dost\k{e}p do specjalist\'ow bez skierowania,
  \item lekarz POZ w ci\k{a}gu 1 dnia roboczego, specjalista w ci\k{a}gu 5 dni,
  \item e-wizyty: konsultacje, skierowania, e-recepty, e-zwolnienia i omawianie wynik\'ow.
\end{itemize}

%--- 3 ---
\noindent{\color{brand}\bfseries 3.~Rehabilitacja i wsparcie po wypadkach}
\begin{itemize}
  \item rehabilitacja powypadkowa ze sprz\k{e}tem w cenie pakietu,
  \item organizacja pomocy po hospitalizacji: piel\k{e}gniarka, pomoc domowa, zwrot koszt\'ow lek\'ow,
  \item dodatkowa rehabilitacja b\'olowa, cz\k{e}sto przydatna przy pracy biurowej i przeci\k{a}\.zeniach kr\k{e}gos\l{}upa.
\end{itemize}

%--- 4 ---
\noindent{\color{brand}\bfseries 4.~Elastyczno\'s\'c i dopasowanie}
\begin{itemize}
  \item mo\.zliwo\'s\'c zmiany pakietu na wy\.zszy w ka\.zdej chwili,
  \item cz\k{e}\'sciowa refundacja koszt\'ow z plac\'owek spoza listy,
  \item model \l{}atwo dopasowa\'c do potrzeb firmy i struktury zespo\l{}u.
\end{itemize}

%--- 5 ---
\noindent{\color{brand}\bfseries 5.~Pe\l{}na obs\l{}uga i opieka doradcy}
\begin{itemize}
  \item bezpo\'sredni kontakt z doradc\k{a} na ka\.zdym etapie,
  \item wsparcie przy wdra\.zaniu, deklaracjach i bie\.z\k{a}cych pytaniach,
  \item organizacja akcji wellbeingowych i edukacyjnych, aby pracownicy faktycznie korzystali z pakietu.
\end{itemize}

%--- 6 ---
\noindent{\color{brand}\bfseries 6.~Pakiet Rodzic}
\begin{itemize}
  \item specjalny, dedykowany pakiet ze specjalistami dla starszych os\'ob,
  \item rozwi\k{a}zanie przydatne tam, gdzie pracownicy cz\k{e}sto wspieraj\k{a} rodzic\'ow lub opiekun\'ow.
\end{itemize}
